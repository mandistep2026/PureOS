# PureOS Demo Script
# This script demonstrates the new features in PureOS v1.1

echo "=== Welcome to PureOS v1.1 Demo ==="
echo ""
echo "Testing new features..."
echo ""

# Create a test directory
mkdir /home/demo
cd /home/demo

# Create some files
echo "Hello from PureOS!" > greeting.txt
echo "This is a test file." > test.txt
echo "Line 1" > data.txt
echo "Line 2" >> data.txt
echo "Line 3" >> data.txt
echo "Line 4" >> data.txt
echo "Line 5" >> data.txt

# List files
echo "Created files:"
ls -l

# Test grep
echo ""
echo "Testing grep command:"
grep "test" test.txt

# Test head/tail
echo ""
echo "Testing head command (first 2 lines):"
head -n2 data.txt

echo ""
echo "Testing tail command (last 2 lines):"
tail -n2 data.txt

# Show command history
echo ""
echo "Command history:"
history

# Show system info
echo ""
echo "System information:"
uname -a
free
df

# Show current directory
echo ""
echo "Current directory:"
pwd

echo ""
echo "=== Demo Complete ==="
echo "Your changes will be saved when you exit."
echo "Run 'exit' to shutdown and save state."
